    for archetype in archetype_performance:
        archetype_performance[archetype]['win_rate'] = (
            archetype_performance[archetype]['winning_trades'] / 
            archetype_performance[archetype]['total_trades']
        )
        archetype_performance[archetype]['avg_curvature'] /= (
            archetype_performance[archetype]['total_trades']
        )
    
    return archetype_performance
